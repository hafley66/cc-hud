```rust
fn render_session_events(&self, ui: &mut egui::Ui, si: usize) {
    let s = &self.data.sessions[si];
    let bg = egui::Color32::from_rgba_premultiplied(25, 22, 17, 255);
    let rect = ui.max_rect();

    // Background for full-width expansion
    ui.painter().rect_filled(rect, 0.0, bg);

    // Event grid header
    let header_y = rect.top() + 5.0;
    ui.painter().text(
        egui::pos2(rect.left() + 20.0, header_y),
        egui::Align2::LEFT_TOP,
        "Time       │ Type │ Model      │ Tool        │ Skill    │ Cost  ",
        egui::FontId::monospace(10.0),
        egui::Color32::from_rgba_premultiplied(160, 150, 130, 255),
    );

    // Event grid header line
    ui.painter().line_segment(
        [
            egui::pos2(rect.left() + 20.0, header_y + 16.0),
            egui::pos2(rect.right() - 20.0, header_y + 16.0),
        ],
        egui::Stroke::new(1.0, egui::Color32::from_rgba_premultiplied(60, 55, 42, 100)),
    );

    // Event rows (egui_table automatically only renders visible rows)
    let y_start = header_y + 24.0;
    let row_h = 18.0;
    let mono = egui::FontId::monospace(9.0);
    let text_col = egui::Color32::from_rgba_premultiplied(200, 190, 165, 200);

    for (ei, event) in s.events.iter().enumerate() {
        let y = y_start + ei as f32 * row_h;
        if y > rect.bottom() - row_h {
            break; // Clip events that overflow
        }

        let x_base = rect.left() + 20.0;

        let (time_str, type_str, model_str, tool_str, skill_str, cost_str) = match event {
            Event::ApiCall { timestamp_secs, model, input_cost_usd, output_cost_usd, .. } => {
                let total_cost = input_cost_usd + output_cost_usd;
                (
                    format_timestamp(*timestamp_secs),
                    "api".to_string(),
                    short_model(model),
                    "-".to_string(),
                    "-".to_string(),
                    format_cost(total_cost),
                )
            }
            Event::ToolUse { name, .. } => (
                "-".to_string(),
                "tool".to_string(),
                "-".to_string(),
                name.clone(),
                "-".to_string(),
                "-".to_string(),
            ),
            Event::SkillUse { skill, .. } => (
                "-".to_string(),
                "skill".to_string(),
                "-".to_string(),
                "-".to_string(),
                skill.clone(),
                "-".to_string(),
            ),
            Event::ReadFile { category, .. } => (
                "-".to_string(),
                "read".to_string(),
                "-".to_string(),
                category.clone(),
                "-".to_string(),
                "-".to_string(),
            ),
            Event::AgentSpawn { subagent_type, description, .. } => (
                "-".to_string(),
                "agent".to_string(),
                "-".to_string(),
                format!("{} {}", subagent_type, description),
                "-".to_string(),
                "-".to_string(),
            ),
            Event::Compaction { .. } => (
                "-".to_string(),
                "compact".to_string(),
                "-".to_string(),
                "-".to_string(),
                "-".to_string(),
                "-".to_string(),
            ),
        };

        // Render columns with fixed positions
        ui.painter().text(
            egui::pos2(x_base, y + 4.0),
            egui::Align2::LEFT_CENTER,
            &time_str,
            mono,
            text_col,
        );
        ui.painter().text(
            egui::pos2(x_base + 90.0, y + 4.0),
            egui::Align2::LEFT_CENTER,
            &type_str,
            mono,
            text_col,
        );
        ui.painter().text(
            egui::pos2(x_base + 150.0, y + 4.0),
            egui::Align2::LEFT_CENTER,
            &model_str,
            mono,
            text_col,
        );
        ui.painter().text(
            egui::pos2(x_base + 280.0, y + 4.0),
            egui::Align2::LEFT_CENTER,
            &tool_str,
            mono,
            text_col,
        );
        ui.painter().text(
            egui::pos2(x_base + 420.0, y + 4.0),
            egui::Align2::LEFT_CENTER,
            &skill_str,
            mono,
            text_col,
        );
        ui.painter().text(
            egui::pos2(x_base + 520.0, y + 4.0),
            egui::Align2::LEFT_CENTER,
            &cost_str,
            mono,
            text_col,
        );
    }
}

fn format_timestamp(secs: u64) -> String {
    let h = (secs % 86400) / 3600;
    let m = (secs % 3600) / 60;
    let s = secs % 60;
    format!("{:02}:{:02}:{:02}", h, m, s)
}

fn short_model(model: &str) -> String {
    if model.len() > 15 {
        &model[..12]
    } else {
        model
    }.to_string()
}

fn format_cost(cost: f64) -> String {
    if cost < 0.001 {
        "-".to_string()
    } else if cost < 0.01 {
        format!("{:.3}", cost)
    } else {
        format!("{:.2}", cost)
    }
}
```

## Notes on Virtualization

egui_table automatically virtualizes rows:
- Only renders visible rows
- Provides `PrefetchInfo.visible_rows` in `prepare()` callback
- Handles scrolling internally
- No manual caching needed

Our implementation just adds event rows to the table data structure and lets egui_table handle the rest.

## Design Decisions

1. **Full-width expansion** - Event grid spans full table width for maximum readability
2. **Individual event rows** - One row per event, no grouping - simpler structure
3. **No caching** - egui_table handles virtualization automatically
4. **Fixed column positions** - Painter-based rendering for precise alignment
5. **Clip to visible area** - Events beyond view are not rendered

---

## Risks

1. **Painter-based rendering** - Event grid uses painter directly instead of egui widgets. May not support some features (copy, select).
2. **Fixed column widths** - Tool/skill columns have limited space. Long names may be truncated.
3. **Variable row heights** - SessionExpanded rows have dynamic height based on event count. egui_table handles this via `row_top_offset()`.
4. **egui_table version** - Library may have breaking updates. Pin to 0.7.

---

## Timeline Column Retention

The timeline column (col 4, 120px wide) shows session activity over time. This is **kept** as requested. The new event grid shows detailed per-event data when a session is expanded.
