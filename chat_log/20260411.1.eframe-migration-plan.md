# Egui Migration Plan: egui_overlay → eframe

## Research Summary

### What I Learned from Documentation

#### egui_overlay Issues
- **Current version**: 0.9.0 (last updated Feb 2024)
- **Dependency lock**: Hard-coded to egui 0.29.1 internally
- **Purpose**: Transparent overlay windows with input passthrough
- **Problem**: Your app already disabled transparency (`transparent_window: Some(false)`), so you're not using the overlay feature

#### eframe (Official egui Framework)
- **Latest version**: 0.34.1 (actively maintained)
- **Dependencies**: Uses latest egui 0.34.1
- **Purpose**: Standard app framework for egui applications
- **Key trait**: `eframe::App` with `ui(&mut self, ui: &mut egui::Ui, frame: &mut eframe::Frame)`
- **Entry point**: `eframe::run_native(app_name, native_options, app_creator)`

#### Key API Differences

**egui_overlay pattern**:
```rust
impl EguiOverlay for Hud {
    fn gui_run(&mut self, egui_context: &egui::Context, backend: &mut GlfwBackend) {
        // Manual window management
        // Manual visibility handling
    }
}

start_overlay(Hud { ... }).enter_event_loop();
```

**eframe pattern**:
```rust
impl App for Hud {
    fn ui(&mut self, ui: &mut egui::Ui, frame: &mut eframe::Frame) {
        // Just UI code - window handled by eframe
    }
}

eframe::run_native("cc-hud", options, Box::new(|cc| Ok(Box::new(Hud::new(cc)))))?;
```

#### eframe NativeOptions
```rust
let options = eframe::NativeOptions {
    centered: true,              // Center window on screen
    resizable: true,             // Allow user to resize
    initial_window_size: Some(egui::vec2(960.0, 460.0)),
    ..Default::default()
};
```

#### App Trait Methods
- **Required**: `ui(&mut self, ui: &mut egui::Ui, frame: &mut eframe::Frame)`
- **Optional**: `save(&mut self, storage: &mut dyn Storage)` - for persistence
- **Optional**: `on_exit(&mut self, gl: Option<&Context>)` - cleanup
- **Optional**: `clear_color(&self, visuals: &Visuals) -> [f32; 4]` - background color

fn main() -> eframe::Result {
    use tracing_subscriber::{fmt, prelude::*, EnvFilter};
    let log_file = std::fs::File::create("/tmp/cc-hud.log").expect("could not create log file");
    tracing_subscriber::registry()
        .with(fmt::layer().with_writer(log_file))
        .with(EnvFilter::try_from_default_env().unwrap_or(EnvFilter::new("info,wgpu=warn,naga=warn")))
        .init();

    let args: Vec<String> = std::env::args().collect();
    let show_history = !args.iter().any(|a| a == "--no-history" || a == "-N");
    let do_backup = args.iter().any(|a| a == "--backup");

    // Backup logic (unchanged)
    if do_backup {
        std::thread::spawn(|| {
            if std::process::Command::new("which")
                .arg("rsync")
                .output()
                .map(|o| o.status.success())
                .unwrap_or(false)
            {
                let dest = format!(
                    "{}/.cc-hud-backup/projects",
                    std::env::var("HOME").unwrap_or_default()
                );
                let _ = std::fs::create_dir_all(&dest);
                let src = format!(
                    "{}/.claude/projects/",
                    std::env::var("HOME").unwrap_or_default()
                );
                match std::process::Command::new("rsync")
                    .args(["-a", "--ignore-existing", &src, &dest])
                    .status()
                {
                    Ok(s) if s.success() => tracing::info!(dest, "backup complete"),
                    Ok(s) => tracing::warn!(code = ?s.code(), "rsync exited non-zero"),
                    Err(e) => tracing::warn!(%e, "rsync failed"),
                }
            } else {
                tracing::warn!("--backup: rsync not found, skipping");
            }
        });
    }

    // Create HUD data
    let hud_data = Arc::new(Mutex::new(HudData::default()));
    let usage_data = Arc::new(Mutex::new(usage::UsageData::default()));

    let feed_data = hud_data.clone();
    std::thread::spawn(move || {
        agent_harnesses::claude_code::poll_loop(feed_data, show_history);
    });

    let feed_usage = usage_data.clone();
    std::thread::spawn(move || {
        usage::poll_loop(feed_usage, Duration::from_secs(90));
    });

    // Set up eframe options
    let native_options = eframe::NativeOptions {
        centered: true,
        resizable: true,
        initial_window_size: Some(egui::vec2(960.0, 460.0)),
        ..Default::default()

    eframe::run_native(
        "cc-hud",
        native_options,
        Box::new(|cc| Ok(Box::new(Hud::new(cc, hud_data, usage_data)))),
    )

impl EguiOverlay for Hud {
    fn gui_run(&mut self, egui_context: &egui::Context, backend: &mut GlfwBackend) {
        let rect = *self.state.lock().unwrap();
        let is_visible = self.visible.load(std::sync::atomic::Ordering::Relaxed);

        if !is_visible {
            glfw_backend.window.set_pos(-9999, -9999);
            glfw_backend.set_window_size([1.0, 1.0]);
            egui_context.request_repaint();
            return;
        }

        if self.first_frame {
            self.first_frame = false;
            glfw_backend.window.set_pos(rect.x, rect.y);
            glfw_backend.window.set_size(rect.w as i32, rect.h as i32);
            glfw_backend.window.show();
        }

        // ... UI code ...
    }
}

impl App for Hud {
    fn ui(&mut self, ui: &mut egui::Ui, frame: &mut eframe::Frame) {
        let data = self.hud_data.lock().unwrap().clone();
        let bg = egui::Color32::from_rgb(14, 12, 9);

        // Remove keyboard shortcuts for visibility (handled by OS)
        // Remove window positioning logic (handled by eframe)

        egui::CentralPanel::default()
            .frame(egui::Frame::NONE.fill(bg))
            .show_inside(ui, |ui| {
                if data.sessions.is_empty() {
                    let area = ui.available_rect_before_wrap();
                    ui.painter().text(
                        area.center(),
                        egui::Align2::CENTER_CENTER,
                        "no active sessions",
                        egui::FontId::monospace(16.0),
                        Palette::TEXT_DIM,
                    );
                    ui.ctx().request_repaint();
                    return;
                }

                // ... rest of existing UI code (unchanged) ...
                let filter_set = match self.filter_mode {
                    FilterMode::Exclude => &mut self.exclude_set,
                    FilterMode::Include => &mut self.include_set,
                };
                // ... existing code continues unchanged ...
            });
    }

    fn save(&mut self, storage: &mut dyn eframe::Storage) {
        // Save billing config
        if let Ok(value) = serde_json::to_value(&self.billing) {
            if let Some(map) = value.as_object() {
                eframe::set_value(storage, eframe::APP_KEY, map);
            }
        }
    }

    fn on_exit(&mut self, _gl: Option<&eframe::glow::Context>) {
        // Save billing config on exit
        self.billing.save_to_file();
    }

    fn clear_color(&self, _visuals: &egui::Visuals) -> [f32; 4] {
        // Background color (dark theme)
        egui::Color32::from_rgb(14, 12, 9).to_normalized_gamma_f32()
    }
}

        if let Some(parent) = Self::config_path().parent() {

        if let Ok(json) = serde_json::to_string_pretty(self) {
